function renderCard(doc) { const data = doc.data(); if (!data.clientName && !data.name) return; const id = doc.id; const path = doc.ref.path; if (document.getElementById('card-' + id)) return; const listDiv = document.getElementById("list"); if (listDiv.innerHTML.includes("Loading")) listDiv.innerHTML = ""; // Time aur Date ko highlight karne ke liye const bookingTime = data.bookingTime || data.time || "Time Not Set"; const bookingDate = data.bookingDate || data.date || "Date Not Set"; const card = document.createElement('div'); card.className = 'card'; card.id = 'card-' + id; card.innerHTML = `
👤 ${data.clientName || data.name} 🕒 ${bookingTime}
📞 ${data.clientPhone || 'N/A'}
✂️ ${data.serviceName || 'N/A'}
📅 Date: ${bookingDate}
History: ${data.history || "No record."}
`; listDiv.appendChild(card); }
