// Ye function check karega ki kaunsa slot dikhana hai aur kaunsa nahi function filterTimeSlots(selectedDate) { const now = new Date(); const today = new Date().toISOString().split('T')[0]; // Aaj ki date const currentHour = now.getHours(); // Abhi kitne baje hain // Database se check karein ki is date par kaunse slots booked hain db.collectionGroup("Bookings").where("bookingDate", "==", selectedDate).get().then((querySnapshot) => { let bookedSlots = []; querySnapshot.forEach((doc) => { bookedSlots.push(doc.data().bookingTime); }); const allSlots = ["10:00 AM", "11:00 AM", "12:00 PM", "01:00 PM", "02:00 PM", "03:00 PM", "04:00 PM", "05:00 PM", "06:00 PM", "07:00 PM"]; let slotHtml = ""; allSlots.forEach(slot => { // Slot ke ghante nikalna (e.g., "10:00 AM" -> 10) let hour = parseInt(slot.split(':')[0]); if (slot.includes("PM") && hour !== 12) hour += 12; const isPast = (selectedDate === today && hour <= currentHour); const isBooked = bookedSlots.includes(slot); if (isPast || isBooked) { // Agar time nikal gaya ya booked hai toh button disable kar do slotHtml += ``; } else { // Agar available hai toh click karne do slotHtml += ``; } }); document.getElementById('slot-container').innerHTML = slotHtml; }); }
